## Data Structure Overview

### TypeScript Interfaces

```typescript
interface Project {
  id: string;
  slug: string;
  title: string;
  tagline: string;
  domain: 'finance' | 'healthcare' | 'retail' | 'nlp' | 'cv' | 'hr';
  industry: string[];
  keyMetric: {
    label: string;
    value: string;
    context: string;
  };
  businessImpact: {
    roi: string;
    savings?: string;
    accuracy?: string;
    other?: { label: string; value: string }[];
  };
  techStack: string[];
  modelType: string;
  overview: string;
  challenge: string;
  solution: string;
  results: string;
  thumbnail: string;
  images: string[];
  demoUrl?: string;
  githubUrl: string;
  seo: {
    metaTitle: string;
    metaDescription: string;
    keywords: string[];
  };
  featured: boolean;
  order: number;
  date: string;
}
```
